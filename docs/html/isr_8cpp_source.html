165 void irqHandler(registers_t r) {
166  /* After every interrupt we need to send an EOI to the PICs
167  * or they will not send another interrupt again */
168 
169  /* slave */
170  if (r.interruptNumber >= 40) {
171  Ports::setPortByte(0xA0, 0x20);
172  }
173  /* master */
174  Ports::setPortByte(0x20, 0x20);
175 
176  /* Handle the interrupt in a more modular way */
177  if (interruptHandlers[r.interruptNumber] != 0) {
178  isr_t handler = interruptHandlers[r.interruptNumber];
179  handler(r);
180  }
181 }
