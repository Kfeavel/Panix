4 disk_load:
5  pusha
6  ; reading from disk requires setting specific values in all registers
7  ; so we will overwrite our input parameters from 'dx'. Let's save it
8  ; to the stack for later use.
9  push dx
10 
11  mov ah, 0x02 ; ah <- int 0x13 function. 0x02 = 'read'
12  mov al, dh ; al <- number of sectors to read (0x01 .. 0x80)
13  ; dh <-- head number (0-15 dec, 0x01--0x80)
14  mov cl, 0x02 ; cl <- sector (0x01 .. 0x11)
15  ; cl <- sector number
16  ; 0x01 is our boot sector, 0x02 is the first 'available' sector
17  mov ch, 0x00 ; ch <- cylinder (0x0 .. 0x3FF, upper 2 bits in 'cl')
18  ; dl <- drive number. Our caller sets it as a parameter and gets it from BIOS
19  ; (0 = floppy, 1 = floppy2, 0x80 = hdd, 0x81 = hdd2)
20  mov dh, 0x00 ; dh <- head number (0x0 .. 0xF)
21 
22  ; [es:bx] <- pointer to buffer where the data will be stored
23  ; caller sets it up for us, and it is actually the standard location for int 13h
24  int 0x13 ; BIOS interrupt (drive test - fail = jmp to disk_error)
25  jc disk_error ; if error (stored in the carry bit)
26

28  cmp al, dh ; BIOS also sets 'al' to the # of sectors read. Compare it.
